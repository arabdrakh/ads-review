        <h1>#️⃣ Hash Tables & Binary Search Trees</h1>
        <h2>Week 5: Hashing and Tree Structures</h2>

        <div class="section">
            <h2>Hashing</h2>
            <p>Hashing maps data to an integer value (hash code) that determines its position in an array.</p>
            
            <div class="key-point">
                <strong>Goal:</strong> Achieve O(1) average-case search, insert, and delete!
            </div>

            <h3>Hash Function Example</h3>
<pre>
// Simple hash for strings
int hash(String key, int tableSize) {
    int hashValue = 0;
    for (char c : key.toCharArray()) {
        hashValue = (hashValue * 31 + c) % tableSize;
    }
    return hashValue;
}

// Example: hash("Mir") with tableSize = 7
// M=77, i=105, r=114
// (77 + 105 + 114) % 7 = 296 % 7 = 2
</pre>
        </div>

        <div class="section">
            <h2>Collision Resolution</h2>
            <p>A <strong>collision</strong> occurs when two different keys hash to the same index.</p>

            <h3>Method 1: Linear Probing</h3>
            <p>If position is occupied, try next position (i+1, i+2, ...)</p>
<pre>
// Insert with linear probing
void insert(K key, V value) {
    int index = hash(key);
    while (table[index] != null) {
        index = (index + 1) % tableSize;  // Move to next
    }
    table[index] = new Entry(key, value);
}
</pre>

            <h3>Method 2: Chaining</h3>
            <p>Each table position holds a linked list of entries.</p>
<pre>
// Array of LinkedLists
LinkedList&lt;Entry&gt;[] table = new LinkedList[tableSize];

void insert(K key, V value) {
    int index = hash(key);
    if (table[index] == null) {
        table[index] = new LinkedList&lt;&gt;();
    }
    table[index].add(new Entry(key, value));
}
</pre>

            <h3>Complexity</h3>
            <table>
                <tr>
                    <th>Case</th>
                    <th>Search/Insert/Delete</th>
                </tr>
                <tr>
                    <td>Average (few collisions)</td>
                    <td><span class="complexity good">O(1)</span></td>
                </tr>
                <tr>
                    <td>Worst (many collisions)</td>
                    <td><span class="complexity bad">O(n)</span></td>
                </tr>
            </table>
        </div>

        <hr>

        <div class="section">
            <h2>Binary Tree</h2>
            <p>Each node has at most <strong>two children</strong> (left and right).</p>

<pre>
class TreeNode&lt;T&gt; {
    T data;
    TreeNode&lt;T&gt; left;
    TreeNode&lt;T&gt; right;
    
    TreeNode(T data) {
        this.data = data;
        this.left = null;
        this.right = null;
    }
}
</pre>

            <h3>Tree Terminology</h3>
            <ul>
                <li><strong>Root:</strong> Top node (no parent)</li>
                <li><strong>Leaf:</strong> Node with no children</li>
                <li><strong>Level/Depth:</strong> Number of edges from root</li>
                <li><strong>Height:</strong> Number of levels in tree</li>
            </ul>
        </div>

        <div class="section">
            <h2>Binary Search Tree (BST)</h2>
            <p>A binary tree where for each node:</p>
            <ul>
                <li>All values in <strong>left subtree</strong> are <strong>less than</strong> node's value</li>
                <li>All values in <strong>right subtree</strong> are <strong>greater than</strong> node's value</li>
            </ul>

<pre>
        8
       / \
      3   10
     / \    \
    1   6    14
       / \   /
      4   7 13
</pre>
        </div>

        <div class="section">
            <h2>Tree Traversals</h2>
            
            <table>
                <tr>
                    <th>Traversal</th>
                    <th>Order</th>
                    <th>Result (above tree)</th>
                </tr>
                <tr>
                    <td><strong>PreOrder</strong></td>
                    <td>Root → Left → Right</td>
                    <td>8, 3, 1, 6, 4, 7, 10, 14, 13</td>
                </tr>
                <tr>
                    <td><strong>InOrder</strong></td>
                    <td>Left → Root → Right</td>
                    <td>1, 3, 4, 6, 7, 8, 10, 13, 14</td>
                </tr>
                <tr>
                    <td><strong>PostOrder</strong></td>
                    <td>Left → Right → Root</td>
                    <td>1, 4, 7, 6, 3, 13, 14, 10, 8</td>
                </tr>
            </table>

            <div class="key-point">
                <strong>Remember:</strong> InOrder traversal of BST gives <strong>sorted order</strong>!
            </div>

            <h3>Traversal Code</h3>
<pre>
// InOrder Traversal (gives sorted output for BST)
void inOrder(TreeNode node) {
    if (node == null) return;
    
    inOrder(node.left);      // Visit left subtree
    System.out.print(node.data + " ");  // Visit root
    inOrder(node.right);     // Visit right subtree
}

// PreOrder Traversal
void preOrder(TreeNode node) {
    if (node == null) return;
    
    System.out.print(node.data + " ");  // Visit root first
    preOrder(node.left);
    preOrder(node.right);
}

// PostOrder Traversal
void postOrder(TreeNode node) {
    if (node == null) return;
    
    postOrder(node.left);
    postOrder(node.right);
    System.out.print(node.data + " ");  // Visit root last
}
</pre>
