        <div class="section">
            <h2>Bubble Sort</h2>
            <p>Repeatedly compares adjacent elements and swaps if in wrong order.</p>
            
            <div class="key-point">
                <strong>Key Idea:</strong> After each pass, the largest unsorted element "bubbles up" to its correct position.
            </div>

<pre>
void bubbleSort(int[] arr) {
    int n = arr.length;
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                // Swap
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}
</pre>
        </div>

        <div class="section">
            <h2>Divide and Conquer</h2>
            <p>A strategy that breaks problems into smaller subproblems, solves them recursively, then combines solutions.</p>
            <ol>
                <li><strong>Divide:</strong> Split problem into smaller parts</li>
                <li><strong>Conquer:</strong> Solve each part recursively</li>
                <li><strong>Combine:</strong> Merge solutions together</li>
            </ol>
        </div>

        <div class="section">
            <h2>Merge Sort</h2>
            <p>Divides array in half recursively, then merges sorted halves.</p>

<pre>
void mergeSort(int[] arr, int left, int right) {
    if (left < right) {
        int mid = (left + right) / 2;
        
        mergeSort(arr, left, mid);       // Sort left half
        mergeSort(arr, mid + 1, right);  // Sort right half
        merge(arr, left, mid, right);    // Merge halves
    }
}

void merge(int[] arr, int left, int mid, int right) {
    // Create temp arrays
    int n1 = mid - left + 1;
    int n2 = right - mid;
    int[] L = new int[n1];
    int[] R = new int[n2];
    
    // Copy data
    for (int i = 0; i < n1; i++) L[i] = arr[left + i];
    for (int j = 0; j < n2; j++) R[j] = arr[mid + 1 + j];
    
    // Merge back
    int i = 0, j = 0, k = left;
    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) {
            arr[k++] = L[i++];
        } else {
            arr[k++] = R[j++];
        }
    }
    
    // Copy remaining
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];
}
</pre>

            <div class="note">
                💡 Merge Sort is <strong>stable</strong> and always O(n log n), but uses O(n) extra space.
            </div>
        </div>

        <div class="section">
            <h2>Quick Sort</h2>
            <p>Selects a pivot, partitions array around it (smaller left, larger right), then recursively sorts.</p>

<pre>
void quickSort(int[] arr, int low, int high) {
    if (low < high) {
        int pi = partition(arr, low, high);
        
        quickSort(arr, low, pi - 1);   // Before pivot
        quickSort(arr, pi + 1, high);  // After pivot
    }
}

int partition(int[] arr, int low, int high) {
    int pivot = arr[high];  // Choose last element as pivot
    int i = low - 1;
    
    for (int j = low; j < high; j++) {
        if (arr[j] < pivot) {
            i++;
            swap(arr, i, j);
        }
    }
    swap(arr, i + 1, high);
    return i + 1;
}
</pre>

            <div class="warning">
                ⚠️ <strong>Worst case O(n²)</strong> occurs when pivot is always smallest or largest element (e.g., already sorted array with last element as pivot).
            </div>
        </div>

        <div class="section">
            <h2>Heap Sort</h2>
            <p>Builds a max heap, then repeatedly extracts the maximum.</p>
            <ol>
                <li>Build max heap from array</li>
                <li>Swap root (max) with last element</li>
                <li>Reduce heap size and heapify root</li>
                <li>Repeat until sorted</li>
            </ol>

<pre>
void heapSort(int[] arr) {
    int n = arr.length;
    
    // Build max heap
    for (int i = n / 2 - 1; i >= 0; i--) {
        heapify(arr, n, i);
    }
    
    // Extract elements one by one
    for (int i = n - 1; i > 0; i--) {
        swap(arr, 0, i);       // Move root to end
        heapify(arr, i, 0);    // Heapify reduced heap
    }
}
</pre>
